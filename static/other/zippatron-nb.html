with open('data.2.csv', 'r') as f:
    text = f.read()
keys = text.split('\n')[0].split(',')[:-1] # get first row split by commas, exclude 
print(f"{keys=}")
DATA = []
for line in text.split("\n")[1:]:
    values = line.split(',')[:-1]
    if len(values) != len(keys):
        print(f"skipping line: '{line}'")
        continue
    dp = {}
    for i in range(len(values)):
        v = values[i]
        try:
            v = float(v)
        except ValueError:
            pass
        dp[keys[i]]=v
    
    DATA.append(dp)
print(f"{len(DATA)=}")
